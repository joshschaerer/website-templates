Tab-Templates

A tab component allows simple navigation through content within a limited space. This page offers a diverse collection of tab templates, designed to meet the needs of various web design projects. Browse through the options and find the perfect fit for your website.

Basic tab

A template for a very simple tab.

Code

                      
                        <!--=============== BASIC TAB ===============-->
                        <div class="tab tab--basic" id="tab-basic">
                          <div class="tab__list" role="tablist">
                            <h3 class="tab__header">
                              <button
                                class="tab__title"
                                id="tab-basic-header-1"
                                role="tab"
                                aria-controls="tab-basic-panel-1"
                                aria-selected="true"
                                tabindex="0">
                                <!-- TITLE -->
                              </button>
                            </h3>
                        
                            <h3 class="tab__header">
                              <button
                                class="tab__title"
                                id="tab-basic-header-2"
                                role="tab"
                                aria-controls="tab-basic-panel-2"
                                aria-selected="false"
                                tabindex="-1">
                                <!-- TITLE -->
                              </button>
                            </h3>
                          </div>
                        
                          <section
                            class="tab__panel"
                            id="tab-basic-panel-1"
                            role="tabpanel"
                            aria-labelledby="tab-basic-header-1"
                            tabindex="-1">
                            <!-- CONTENT -->
                          </section>
                        
                          <section
                            class="tab__panel"
                            id="tab-basic-panel-2"
                            role="tabpanel"
                            aria-labelledby="tab-basic-header-2"
                            tabindex="-1"
                            hidden>
                            <!-- CONTENT -->
                          </section>
                        </div>
                      
                    
                      
                        /*=============== VARIABLES CSS ===============*/
                        :root {
                          --hover-overlay-color: rgba(0, 0, 0, 0.05);
                          --active-overlay-color: hsl(131, 100%, 95%);
                          --first-color: hsl(135, 48%, 51%);
                          --first-color-alt: hsl(135, 89%, 52%, 0.4);
                        }

                        /*=============== TAB ===============*/
                        [role="tablist"] {
                          display: flex;
                          flex-direction: row;
                          flex-wrap: wrap;
                        }

                        [role="tab"] {
                          padding: 0.5rem;
                          background-color: transparent;
                        }

                        [role="tab"]:is(:hover, :focus) {
                          background-color: var(--hover-overlay-color);
                        }

                        [role="tab"][aria-selected="true"] {
                          background-color: var(--active-overlay-color);
                          color: var(--first-color);
                        }

                        [role="tab"][aria-selected="true"]:is(:hover, :focus) {
                          background-color: var(--first-color-alt);
                        }

                        [role="tabpanel"] {
                          margin-top: 0.5rem;
                        }
                      
                    
                      
                        /*=============== TAB ===============*/
                        const tabLists = document.querySelectorAll('[role="tablist"]');

                        // Setup each tablist
                        tabLists.forEach((tabList) => {
                          const tabs = tabList.querySelectorAll('[role="tab"]');
                          let tabFocus = 0;

                          // Add a keydown event handler to the tablist
                          tabList.addEventListener("keydown", (e) => {
                            if (e.key === "ArrowRight" || e.key === "ArrowLeft") {
                              tabs[tabFocus].setAttribute("tabindex", -1);
                              // Move right
                              if (e.key === "ArrowRight") {
                                tabFocus++;
                                // If we're at the end, go to the start
                                if (tabFocus >= tabs.length) {
                                  tabFocus = 0;
                                }
                                // Move left
                              } else if (e.key === "ArrowLeft") {
                                tabFocus--;
                                // If we're at the start, move to the end
                                if (tabFocus < 0) {
                                  tabFocus = tabs.length - 1;
                                }
                              }

                              tabs[tabFocus].setAttribute("tabindex", 0);
                              tabs[tabFocus].focus();
                            }
                          });

                          // Add a click event handler to each tab
                          tabs.forEach((tab) => {
                            tab.addEventListener("click", changeTabs);
                          });
                        });

                        function changeTabs(e) {
                          const target = e.target;
                          const parent = target.parentNode.parentNode;
                          const grandparent = parent.parentNode;

                          // Remove all current selected tabs
                          parent
                            .querySelectorAll('[aria-selected="true"]')
                            .forEach((tab) => tab.setAttribute("aria-selected", false));

                          // Set this tab as selected
                          target.setAttribute("aria-selected", true);

                          // Hide all tab panels
                          grandparent
                            .querySelectorAll('[role="tabpanel"]')
                            .forEach((panel) => panel.setAttribute("hidden", true));

                          // Show the selected panel
                          grandparent
                            .querySelector(`#${target.getAttribute("aria-controls")}`)
                            .removeAttribute("hidden");
                        }
